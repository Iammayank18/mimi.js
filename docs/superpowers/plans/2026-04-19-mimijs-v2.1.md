# mimijs v2.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 7 production-grade components (radix router, lifecycle hooks, RequestContext, ValidatorAdapter+Zod, rate limiting, graceful shutdown, auto OpenAPI) to the existing TypeScript mimijs codebase without breaking any v1 API.

**Architecture:** Keep existing Layer/Route/Router and prototype-patched req/res intact. Add find-my-way as a parallel radix trie lookup for registered routes. Thread a HookRegistry through the pipeline. Add a ctx-style RequestContext that wraps req/res per-request. All new APIs are opt-in.

**Tech Stack:** TypeScript (strict), find-my-way (radix trie), zod + zod-to-json-schema (validation), vitest (tests), existing raw-body/mime-types/swagger-jsdoc.

```typescript
import type { HookFn, HookName, RequestContext } from '../types';

const HOOK_NAMES: HookName[] = [
  'onRequest',
  'preParsing',
  'preValidation',
  'preHandler',
  'onSend',
  'onResponse',
  'onError',
  'onClose',
];

export class HookRegistry {
  private readonly slots = new Map<HookName, HookFn[]>(
    HOOK_NAMES.map((name) => [name, []]),
  );

  addHook(name: HookName, fn: HookFn): void {
    const slot = this.slots.get(name);
    if (!slot) throw new TypeError(`Unknown hook name: "${name}"`);
    slot.push(fn);
  }

  async runHooks(name: HookName, ctx: RequestContext, ...rest: unknown[]): Promise<void> {
    const slot = this.slots.get(name) ?? [];
    for (const fn of slot) {
      await fn(ctx, ...rest);
    }
  }
}
```

```typescript
import crypto from 'crypto';
import type { MimiRequest, MimiResponse, RequestContext } from '../types';

export function createContext(req: MimiRequest, res: MimiResponse): RequestContext {
  const ctx: RequestContext = {
    req,
    res,

    body: req.body as unknown,
    params: req.params,
    query: req.query,

    requestId: crypto.randomUUID(),
    startTime: Date.now(),
    ip: req.ip,
    method: req.method ?? 'GET',
    path: req.path,

    store: {},

    json(data, status) {
      if (status !== undefined) res.statusCode = status;
      res.json(data);
    },

    send(data, status) {
      if (status !== undefined) res.statusCode = status;
      res.send(data);
    },

    html(data, status) {
      if (status !== undefined) res.statusCode = status;
      res.set('Content-Type', 'text/html; charset=utf-8');
      res.send(data);
    },

    redirect(url, status) {
      res.redirect(url, status);
    },

    status(code) {
      res.statusCode = code;
      return ctx;
    },

    set(header, value) {
      res.set(header, value);
      return ctx;
    },

    get(header) {
      return req.get(header);
    },

    is(contentType) {
      return req.is(contentType);
    },
  };

  return ctx;
}
```

```typescript
import { Layer } from './layer';
import type { MimiRequest, MimiResponse, NextFunction, RequestHandler } from '../types';

// Circular reference via lazy require — resolved at runtime
type RadixInstance = import('find-my-way').Instance<import('find-my-way').HTTPVersion.V1>;

const HTTP_METHODS = ['get', 'post', 'put', 'patch', 'delete', 'head', 'options', 'all'] as const;
const ALL_METHODS = ['GET', 'POST', 'PUT', 'PATCH', 'DELETE', 'HEAD', 'OPTIONS'] as const;

export class Route {
  path: string;
  stack: Layer[] = [];
  methods: Record<string, boolean> = {};

  /** Set by Router when it creates a Route; used to register dispatch in the radix trie. */
  _radix?: RadixInstance;

  constructor(path: string) {
    this.path = path;
    this.dispatch = this.dispatch.bind(this);
  }

  _handles_method(method: string): boolean {
    if (this.methods['all']) return true;
    return Boolean(this.methods[method.toLowerCase()]);
  }

  dispatch(req: MimiRequest, res: MimiResponse, done: NextFunction): void {
    let idx = 0;
    const stack = this.stack;
    const method = req.method?.toLowerCase() ?? 'get';

    function next(err?: Error | string): void {
      if (err && err === 'route') return done();
      if (err && err === 'router') return done(err as string);

      const layer = stack[idx++];
      if (!layer) return done(err as Error | undefined);

      if (layer.method && layer.method !== method && method !== 'head') {
        return next(err);
      }

      if (err) {
        layer.handleError(err as Error, req, res, next);
      } else {
        layer.handleRequest(req, res, next);
      }
    }

    next();
  }

  /** Register this route's dispatch in the radix trie for the given HTTP method. */
  private _registerInRadix(method: string): void {
    if (!this._radix) return;
    const methods = method === 'all' ? ALL_METHODS : [method.toUpperCase() as typeof ALL_METHODS[number]];
    for (const m of methods) {
      try {
        this._radix.on(m, this.path, this.dispatch as any);
      } catch {
        // Route already registered (e.g. calling .get() twice) — ignore duplicate
      }
    }
  }
}

// Dynamically add HTTP method shortcuts to Route.prototype
HTTP_METHODS.forEach((method) => {
  (Route.prototype as any)[method] = function (...handlers: RequestHandler[]) {
    handlers.forEach((handler) => {
      const layer = new Layer('/', {}, handler);
      if (method !== 'all') {
        layer.method = method;
        this.methods[method] = true;
      } else {
        this.methods['all'] = true;
      }
      this.stack.push(layer);
    });
    (this as Route)._registerInRadix(method);
    return this;
  };
});
```

```typescript
import FindMyWay from 'find-my-way';
import { Layer } from './layer';
import { Route } from './route';
import { createContext } from '../core/context';
import type {
  MimiRequest, MimiResponse, NextFunction, Middleware, Route as IRoute,
} from '../types';
import type { HookRegistry } from '../core/hooks';

const HTTP_METHODS = ['get', 'post', 'put', 'patch', 'delete', 'head', 'options', 'all'] as const;

export class Router {
  /** Middleware-only layers (registered via use()). */
  stack: Layer[] = [];

  /** Radix trie for O(log n) route matching. */
  readonly radix = FindMyWay<FindMyWay.HTTPVersion.V1>({ ignoreTrailingSlash: true });

  /** Optional hook registry — set by the mimi() application factory. */
  hookRegistry?: HookRegistry;

  use(path: string | Middleware, ...fns: Middleware[]): this {
    let mountPath: string;
    let callbacks: Middleware[];

    if (typeof path === 'function') {
      mountPath = '/';
      callbacks = [path, ...fns];
    } else {
      mountPath = path;
      callbacks = fns;
    }

    callbacks.forEach((fn) => {
      const layer = new Layer(mountPath, { end: false, strict: false }, fn as any);
      layer.route = undefined;
      this.stack.push(layer);
    });

    return this;
  }

  route(path: string): IRoute {
    const route = new Route(path);
    route._radix = this.radix;
    return route as unknown as IRoute;
  }

  handle(req: MimiRequest, res: MimiResponse, done: NextFunction): void {
    let idx = 0;
    const stack = this.stack;
    const originalUrl = req.url ?? '/';
    const hooks = this.hookRegistry;

    const matchRoute = (): void => {
      const match = this.radix.find(req.method as FindMyWay.HTTPMethod ?? 'GET', req.path);
      if (!match) {
        // No route registered — call done() so finalhandler sends 404
        return done(undefined);
      }

      req.params = { ...req.params, ...(match.params as Record<string, string>) };

      const ctx = createContext(req, res);
      (req as any)._mimiCtx = ctx;

      const routeDispatch = match.handler as (req: MimiRequest, res: MimiResponse, next: NextFunction) => void;

      runHookPipeline(ctx, routeDispatch, hooks).catch((err) => done(err as Error));
    };

    const next = (err?: Error | string): void => {
      const layer = stack[idx++];
      if (!layer) return matchRoute();

      const path = req.path ?? '/';
      if (!layer.match(path)) return next(err);

      // Strip mount prefix for sub-router delegation
      if (layer.path !== undefined && layer.path !== '/') {
        req.url = originalUrl.slice(layer.path.length) || '/';
      }

      const restore = (e?: Error | string): void => {
        req.url = originalUrl;
        next(e);
      };

      if (err) {
        layer.handleError(err as Error, req, res, restore);
      } else {
        layer.handleRequest(req, res, restore);
      }
    };

    next();
  }
}

/**
 * Runs the 7-hook pipeline around a route dispatch.
 * Any thrown error skips to onError; if no onError hook handles it,
 * the error is re-thrown so the caller can pass it to done().
 */
async function runHookPipeline(
  ctx: import('../types').RequestContext,
  routeDispatch: (req: MimiRequest, res: MimiResponse, next: NextFunction) => void,
  hooks?: HookRegistry,
): Promise<void> {
  try {
    if (hooks) await hooks.runHooks('onRequest', ctx);
    if (hooks) await hooks.runHooks('preParsing', ctx);
    if (hooks) await hooks.runHooks('preValidation', ctx);
    if (hooks) await hooks.runHooks('preHandler', ctx);

    // Call the route dispatch function; wrap in a Promise so async handlers are awaited.
    await new Promise<void>((resolve, reject) => {
      routeDispatch(ctx.req, ctx.res, (err?: Error | string) => {
        if (err) reject(typeof err === 'string' ? new Error(err) : err);
        else resolve();
      });
    });

    if (hooks) await hooks.runHooks('onSend', ctx);

    // onResponse runs non-blocking after the response is sent
    ctx.res.on('finish', () => {
      if (hooks) hooks.runHooks('onResponse', ctx).catch(() => {});
    });
  } catch (err) {
    if (hooks) {
      await hooks.runHooks('onError', ctx, err as Error);
    }
    // If the error hook didn't write a response, propagate it
    if (!(ctx.res as any).headersSent) {
      throw err;
    }
  }
}

// Dynamically add HTTP method shortcuts to Router.prototype
HTTP_METHODS.forEach((method) => {
  (Router.prototype as any)[method] = function (path: string, ...handlers: Middleware[]) {
    const route = this.route(path);
    (route as any)[method](...handlers);
    return this;
  };
});
```

```typescript
import { pathToRegexp, Key } from 'path-to-regexp';
import type { MimiRequest, MimiResponse, NextFunction, LayerOptions } from '../types';

type Handler = (...args: any[]) => any;

export class Layer {
  regexp: RegExp;
  keys: Key[] = [];
  handle: Handler;
  method?: string;
  route?: import('./route').Route;
  params: Record<string, string> = {};
  path?: string;

  private readonly fastSlash: boolean;
  private readonly fastStar: boolean;

  constructor(path: string, options: LayerOptions, fn: Handler) {
    this.handle = fn;
    this.keys = [];

    const opts = {
      end: options.end ?? true,
      strict: options.strict ?? false,
      sensitive: options.sensitive ?? false,
    };

    this.fastSlash = path === '/' && opts.end === false;
    this.fastStar = path === '*';

    this.regexp = pathToRegexp(path, this.keys, opts);
  }

  match(path: string): boolean {
    if (this.fastSlash) {
      this.params = {};
      this.path = '';
      return true;
    }

    if (this.fastStar) {
      this.params = { '0': decodeParam(path) };
      this.path = path;
      return true;
    }

    const match = this.regexp.exec(path);
    if (!match) {
      this.params = {};
      this.path = undefined;
      return false;
    }

    this.params = {};
    this.path = match[0];

    for (let i = 0; i < this.keys.length; i++) {
      const key = this.keys[i];
      const prop = String(key.name);
      const val = match[i + 1];
      if (val !== undefined) {
        this.params[prop] = decodeParam(val);
      }
    }

    return true;
  }

  handleRequest(req: MimiRequest, res: MimiResponse, next: NextFunction): void {
    const fn = this.handle;
    if (fn.length > 3) {
      // error handler — skip in normal flow
      return next();
    }

    try {
      let result: unknown;

      if (fn.length === 1) {
        // ctx-style handler: async (ctx) => {}
        const ctx = (req as any)._mimiCtx;
        result = fn(ctx);
      } else {
        // Express-style handler: (req, res, next) => {}
        result = fn(req, res, next);
      }

      if (result && typeof (result as any).then === 'function') {
        // Await the promise and call next() when it resolves
        (result as Promise<unknown>).then(() => next(), (err: Error) => next(err));
      }
    } catch (err) {
      next(err as Error);
    }
  }

  handleError(err: Error, req: MimiRequest, res: MimiResponse, next: NextFunction): void {
    const fn = this.handle;
    if (fn.length !== 4) {
      return next(err);
    }
    try {
      fn(err, req, res, next);
    } catch (e) {
      next(e as Error);
    }
  }
}

function decodeParam(val: string): string {
  try {
    return decodeURIComponent(val);
  } catch {
    return val;
  }
}
```

```typescript
import './request';
import './response';
import http from 'http';
import { Router } from '../router';
import { HookRegistry } from './hooks';
import { createFinalHandler } from './finalhandler';
import { createRegister } from '../plugins';
import loadRoutes from '../router-loader';
import type { MimiApp, Middleware, RequestHandler, CtxHandler, Route, HookName, HookFn } from '../types';

export function mimi(): MimiApp {
  const hookRegistry = new HookRegistry();
  const router = new Router();
  router.hookRegistry = hookRegistry;

  let server: http.Server | null = null;
  let inFlight = 0;
  let drainResolve: (() => void) | null = null;

  function app(req: http.IncomingMessage, res: http.ServerResponse): void {
    inFlight++;
    res.on('finish', () => {
      inFlight--;
      if (inFlight === 0 && drainResolve) {
        drainResolve();
        drainResolve = null;
      }
    });
    router.handle(req as any, res as any, createFinalHandler(req, res));
  }

  const mimiApp = app as unknown as MimiApp;

  (mimiApp as any).use = function (
    path: string | Middleware,
    ...fns: Middleware[]
  ): MimiApp {
    router.use(path as any, ...(fns as any[]));
    return mimiApp;
  };

  (mimiApp as any).route = function (path: string): Route {
    return router.route(path);
  };

  const HTTP_METHODS = [
    'get', 'post', 'put', 'patch', 'delete', 'head', 'options', 'all',
  ] as const;

  HTTP_METHODS.forEach((method) => {
    (mimiApp as any)[method] = function (
      path: string,
      ...handlers: (RequestHandler | CtxHandler)[]
    ): MimiApp {
      (router as any)[method](path, ...handlers);
      return mimiApp;
    };
  });

  (mimiApp as any).listen = function (
    port: number,
    callback?: () => void,
  ): http.Server {
    server = http.createServer(app);
    return server.listen(port, callback);
  };

  (mimiApp as any).register = createRegister(mimiApp);

  (mimiApp as any).addHook = function (name: HookName, fn: HookFn): MimiApp {
    hookRegistry.addHook(name, fn);
    return mimiApp;
  };

  (mimiApp as any).close = function (timeoutMs = 10_000): Promise<void> {
    return new Promise((resolve, reject) => {
      if (server) {
        server.close((err) => { if (err) reject(err); });
      }

      const finish = () => {
        hookRegistry.runHooks('onClose', {} as any).then(resolve).catch(reject);
      };

      if (inFlight === 0) {
        finish();
        return;
      }

      drainResolve = finish;

      // Force-close after timeout
      setTimeout(() => {
        drainResolve = null;
        finish();
      }, timeoutMs).unref();
    });
  };

  // Auto-load routes from consumer's ./routes/ directory
  loadRoutes(mimiApp).catch(() => {});

  return mimiApp;
}
```

```typescript
import { ZodError } from 'zod';
import { zodAdapter } from './zod';
import { formatZodError } from './zod';
import type { ValidatorAdapter } from './adapter';
import type { RequestContext } from '../types';

export type { ValidatorAdapter } from './adapter';
export { zodAdapter } from './zod';

let _adapter: ValidatorAdapter = zodAdapter;

/** Replace the global validator adapter (e.g. with a Valibot adapter). */
export function setAdapter(adapter: ValidatorAdapter): void {
  _adapter = adapter;
}

export function getAdapter(): ValidatorAdapter {
  return _adapter;
}

export interface ValidationTargets {
  body?: unknown;
  params?: unknown;
  query?: unknown;
  headers?: unknown;
}

/**
 * Middleware factory that validates ctx.body/params/query against the provided schemas.
 * Responds 400 on failure. Works with both ctx-style and req/res/next handlers.
 */
export function validate(targets: ValidationTargets) {
  return async (ctx: RequestContext, _res: unknown, next: () => void): Promise<void> => {
    const adapter = _adapter;
    try {
      if (targets.body !== undefined) {
        ctx.body = await adapter.parseAsync(targets.body, ctx.req.body);
        ctx.req.body = ctx.body;
      }
      if (targets.params !== undefined) {
        ctx.params = await adapter.parseAsync(targets.params, ctx.req.params) as any;
        ctx.req.params = ctx.params;
      }
      if (targets.query !== undefined) {
        ctx.query = await adapter.parseAsync(targets.query, ctx.req.query) as any;
        ctx.req.query = ctx.query;
      }
      if (targets.headers !== undefined) {
        await adapter.parseAsync(targets.headers, ctx.req.headers);
      }
      next();
    } catch (err) {
      if (err instanceof ZodError) {
        ctx.json({ error: 'Validation failed', issues: formatZodError(err) }, 400);
      } else {
        ctx.json({ error: 'Validation failed', issues: [{ path: [], message: String((err as Error).message) }] }, 400);
      }
    }
  };
}
```

```typescript
import type { RequestContext } from '../types';

export interface RateLimitOptions {
  /** Time window in ms. Default: 60_000 (1 minute). */
  windowMs?: number;
  /** Max requests per window per key. Default: 100. */
  max?: number;
  /** Extract the bucket key from ctx. Default: ctx.ip. */
  keyBy?: (ctx: RequestContext) => string;
  /** Called when limit is exceeded (before 429 response is sent). */
  onExceeded?: (ctx: RequestContext) => void;
}

interface Bucket {
  count: number;
  resetAt: number;
}

export function rateLimit(options: RateLimitOptions = {}) {
  const windowMs = options.windowMs ?? 60_000;
  const max = options.max ?? 100;
  const keyBy = options.keyBy ?? ((ctx) => ctx.ip);

  const store = new Map<string, Bucket>();

  // Periodically clean up expired windows to prevent memory leaks
  const cleanup = setInterval(() => {
    const now = Date.now();
    for (const [key, bucket] of store) {
      if (now >= bucket.resetAt) store.delete(key);
    }
  }, windowMs);

  // Allow the process to exit even if cleanup is still scheduled
  if (cleanup.unref) cleanup.unref();

  return async (ctx: RequestContext, _res: unknown, next: () => void): Promise<void> => {
    const key = keyBy(ctx);
    const now = Date.now();

    let bucket = store.get(key);
    if (!bucket || now >= bucket.resetAt) {
      bucket = { count: 0, resetAt: now + windowMs };
      store.set(key, bucket);
    }

    bucket.count++;

    const remaining = Math.max(0, max - bucket.count);
    const retryAfter = Math.ceil((bucket.resetAt - now) / 1_000);

    ctx.set('X-RateLimit-Limit', String(max));
    ctx.set('X-RateLimit-Remaining', String(remaining));
    ctx.set('X-RateLimit-Reset', String(Math.ceil(bucket.resetAt / 1_000)));

    if (bucket.count > max) {
      ctx.set('Retry-After', String(retryAfter));
      if (options.onExceeded) options.onExceeded(ctx);
      ctx.json({ error: 'Too Many Requests' }, 429);
      return;
    }

    next();
  };
}
```

```typescript
  url?: string;

export interface SwaggerOptions {
  info: {
    title: string;
    version: string;
    description?: string;
    license?: SwaggerLicense;
    contact?: SwaggerContact;
  };
  filesPattern?: string;
  /** When true, merge route schemas (from validate() calls) into the spec. */
  autoSpec?: boolean;

/** A registered route entry used for autoSpec generation. */
export interface RouteSchemaEntry {
  method: string;
  path: string;
  schema?: {
    body?: unknown;
    params?: unknown;
    query?: unknown;
  };
}

/** Build an OpenAPI 3.0.0 spec object from an array of route schema entries. */
export function buildAutoSpec(
  info: { title: string; version: string },
  routes: RouteSchemaEntry[],
): Record<string, unknown> {
  const paths: Record<string, unknown> = {};
  const adapter = getAdapter();

  for (const route of routes) {
    if (!route.schema) continue;
    const pathKey = route.path;
    const methodKey = route.method.toLowerCase();

    if (!paths[pathKey]) paths[pathKey] = {};
    const pathItem = paths[pathKey] as Record<string, unknown>;

    const operation: Record<string, unknown> = { responses: { '200': { description: 'Success' } } };

    if (route.schema.body && adapter.toOpenAPI) {
      operation.requestBody = {
        content: { 'application/json': { schema: adapter.toOpenAPI(route.schema.body) } },
      };
    }

    pathItem[methodKey] = operation;
  }

  return {
    openapi: '3.0.0',
    info: { title: info.title, version: info.version },
    paths,
  };
}

/** Global route registry — populated by setupSwagger when autoSpec is enabled. */
const routeRegistry: RouteSchemaEntry[] = [];

/** Register a route's schema for autoSpec generation. */
export function registerRouteSchema(entry: RouteSchemaEntry): void {
  routeRegistry.push(entry);
}

export function setupSwagger(app: MimiApp, options: SwaggerOptions): void {
  const spec = swaggerJSDoc({
    definition: {
      openapi: '3.0.0',
      info: options.info,
    },
    apis: [options.filesPattern ?? './**/*.js'],
  });

  if (options.autoSpec && routeRegistry.length > 0) {
    const autoSpec = buildAutoSpec(options.info, routeRegistry);
    const autoPaths = (autoSpec.paths ?? {}) as Record<string, unknown>;
    for (const [path, item] of Object.entries(autoPaths)) {
      (spec as any).paths = (spec as any).paths ?? {};
      (spec as any).paths[path] = {
        ...((spec as any).paths[path] ?? {}),
        ...(item as Record<string, unknown>),
      };
    }
  }

  (app as any).get('/api-docs/swagger.json', (_req: any, res: any) => res.json(spec));
  (app as any).get('/api-docs', (_req: any, res: any) => {
    res.set('Content-Type', 'text/html');
    res.send(`<!DOCTYPE html><html><head><title>${options.info.title}</title>
      <link rel="stylesheet" href="https://unpkg.com/swagger-ui-dist@5/swagger-ui.css" />
    </head><body><div id="swagger-ui"></div>
      <script src="https://unpkg.com/swagger-ui-dist@5/swagger-ui-bundle.js"></script>
      <script>SwaggerUIBundle({ url: '/api-docs/swagger.json', dom_id: '#swagger-ui' })</script>
    </body></html>`);
  });
}
```

```typescript
async function request(
  server: http.Server,
  method: string,
  path: string,
  body?: unknown,
): Promise<{ status: number; data: unknown; headers: http.IncomingHttpHeaders }> {
  return new Promise((resolve, reject) => {
    const addr = server.address() as { port: number };
    const payload = body ? JSON.stringify(body) : undefined;
    const req = http.request({
      hostname: '127.0.0.1',
      port: addr.port,
      path,
      method,
      headers: {
        'Content-Type': 'application/json',
        ...(payload ? { 'Content-Length': Buffer.byteLength(payload) } : {}),
      },
    }, (res) => {
      let raw = '';
      res.on('data', (c) => { raw += c; });
      res.on('end', () => {
        let data: unknown;
        try { data = JSON.parse(raw); } catch { data = raw; }
        resolve({ status: res.statusCode ?? 0, data, headers: res.headers });
      });
    });
    req.on('error', reject);
    if (payload) req.write(payload);
    req.end();
  });
}
```
